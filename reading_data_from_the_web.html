get_all <- function(url) {
  out <- NULL
  
  while(!is.null(url)) {
    message("Getting ", url)
    req <- GET(url)
    stop_for_status(req)
    
    con <- content(req)
    out <- c(out, con$results)
    url <- con$`next`
  }
  
  out
}
